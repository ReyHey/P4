function void main() {
	number GCD = gcd(10, 20)
	
	write(gcd.ToText() + "\n")
}
function number gcd(number a, number b) {
	number GCD
	number i
	number smallest

	if a == 0 or b == 0 {
		return 0
	}

	if a <= b {
		smallest = a
	}
	else {
		smallest = b
	}

	i = smallest
	while (a mod i != 0 or b mod i != 0) and i > 0 {
		i = i - 1
	}
	GCD = i

	return GCD
}
